// **REQUIRES CMAKE INSTALLATION (available through msys2)**

// SETTING UP BUILD ENVIRONMENT AND FORCE CLANG(only needed once)
run:
cmake -S . -B build -DCMAKE_TOOLCHAIN_FILE=./clang-toolchain.cmake

// SET UP BUILD ENVIRONMENT IF YOU WANT BUILD TESTS DISABLED TO SPEED UP COMPILE TIMES
cmake -S . -B build -DCMAKE_TOOLCHAIN_FILE=./clang-toolchain.cmake -DASSIMP_BUILD_TESTS=OFF -D GLFW_BUILD_TESTS=OFF



// COMPILING PROJECT:
run: 
cmake --build build --config Release

will produce:
sandbox.exe
